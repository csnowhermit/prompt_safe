3.1 模块定义

模块名：GatewayGuard | 部署：API网关插件（Kong / Envoy / Nginx）

3.2 处理流程

request_phase: 1. IP Blacklist Check - Redis Sorted Set: "ip_blacklist" → SCORE = ban_until_timestamp - 命中黑名单 → HTTP 403 + {"error":"ACCESS_DENIED","code":"IP_BLOCKED"} 2. Rate Limiting (per user / per IP) - 算法: Token Bucket - 普通用户: 60 req/min, Agent: 20 req/min, Admin: 无限制 - 超限 → HTTP 429 + Retry-After header 3. Authentication - Bearer Token / API Key 验证 - JWT claims: {sub, role, exp, iat} - 失败 → HTTP 401 4. WAF Basic Checks - ModSecurity CRS 规则集（SQLi/XSS/PathTraversal） - 命中 → HTTP 400 + {"error":"REQUEST_BLOCKED","rule_id":"..."} 5. Generate TraceID & Inject Headers - X-Trace-ID: UUID (整个链路复用) - X-Session-ID: UUID (首次创建，后续携带) - X-User-Role: end_user|employee|admin|agent

4.3 核心算法流程

# InputGuard 主处理管线
def process(input_text, user_context, mode="chat"):
    ctx = SecurityContext(trace_id=user_context.trace_id)

    # ── Step 1: 基础校验 ──
    if len(input_text) > MAX_CHARS[mode]:
        return reject("INPUT_TOO_LONG")
    if detect_obfuscation(input_text).score > OBFS_THRESHOLD:
        ctx.risk_level = RiskLevel.RED
        return reject("OBFUSCATION_DETECTED")

    # ── Step 2: 注入关键词 & 模式匹配 ──
    keyword_hits = keyword_matcher.scan(input_text)
    pattern_hits = pattern_matcher.scan(input_text)
    if keyword_hits or pattern_hits:
        ctx.add_rules(keyword_hits + pattern_hits)

    # ── Step 3: 安全意图分类 (独立小模型) ──
    classification = safety_classifier.classify(input_text)
    # 模型: DistilBERT fine-tuned on in-house jailbreak dataset
    if classification.intent == "jailbreak":
        return block("JAILBREAK_DETECTED", classification)
    elif classification.intent == "info_extract":
        return block("INFO_EXTRACTION_DETECTED", classification)

    # ── Step 4: PII脱敏 ──
    sanitized = piimasker.mask(input_text)
    # 脱敏规则: 手机号/身份证/邮箱/银行卡/地址/IP/MAC

    # ── Step 5: 标准化 ──
    normalized = normalize(sanitized)  # UTF-8, trim, remove ctrl chars

    # ── 判定风险等级 ──
    risk = determine_risk_level(keyword_hits, pattern_hits, classification)
    return GuardResult(text=normalized, risk=risk, ...)

# 高安全场景：dual_model_verification mode
def dual_model_inference(prompt, context):

    main_task = async_infer(model_a, prompt, SECURITY_PARAMS)
    safety_task = async_infer(model_b, SAFETY_CHECK_PROMPT, SECURITY_PARAMS)

7.3 五步后校验管线

def output_guard_pipeline(raw_output, context):
    # ── Step 1: 规则匹配 ──
    # 检查是否包含System Prompt片段/复述恶意指令/禁止关键词
    for rule in OUTPUT_RULES:
        if rule.matches(raw_output):
            if rule.severity == "critical":
                return block(rule.reason)

    # ── Step 2: 内容安全审核 ──
    content_check = content_moderator.check(raw_output)
    # 检测: violence/sexual/political/hate/fraud/suicide
    if content_check.is_unsafe:
        return block(content_check.category)

    # ── Step 3: PII脱敏 ──
    masked = piimasker.mask(raw_output)
    pii_count = piimasker.last_count

    # ── Step 4: 格式校验 ──
    if len(masked) > MAX_OUTPUT_LEN:
        masked = masked[:MAX_OUTPUT_LEN] + "...（内容过长已截断）"

    # ── Step 5: 裁决分发 ──
    result = {"final_output": masked, "actions": []}
    if pii_count > 0:
        result["decision"] = "masked"
        result["actions"].append("pii_mask")
    if context.input_risk_level == "yellow":
        result["final_output"] += "\n> 以上内容由AI生成，仅供参考。"
        result["actions"].append("append_disclaimer")

    return result

8.3 上下文窗口滑动策略

def sliding_window_truncate(context, max_tokens):
    """
    滑动窗口裁剪策略：
    - System Prompt 始终保留（标记 persistent=true）
    - 从最近一轮开始向前取，直到达到Token上限
    - 完全被裁剪的轮次从历史中移除
    """
    system_msgs = [m for m in context if m.get("persistent")]
    history_msgs = [m for m in context if not m.get("persistent")]

    system_tokens = count_tokens(system_msgs)
    budget = max_tokens - system_tokens

    # 从最近开始累积
    kept = []
    token_sum = 0
    for msg in reversed(history_msgs):
        t = count_tokens(msg)
        if token_sum + t > budget:
            break
        kept.insert(0, msg)
        token_sum += t

    return system_msgs + kept

10.2 文档入库安全管线

def rag_ingestion_pipeline(document, user_context):
    # ── 1. 来源鉴权 ──
    if not source_authenticator.verify(document.source):
        return reject("SOURCE_AUTH_FAILED")

    # ── 2. 文件类型白名单 ──
    ALLOWED_TYPES = [".pdf", ".txt", ".md", ".docx", ".html"]
    if document.filetype not in ALLOWED_TYPES:
        return reject("UNSUPPORTED_FILE_TYPE")

    # ── 3. 文件大小限制 ──
    if document.size > 50 * 1024 * 1024:  # 50MB
        return reject("FILE_TOO_LARGE")

    # ── 4. 沙盒内解析 ──
    with sandbox(timeout=30):
        parsed_text = document_parser.parse(document)
        # 解析结果走输入防护管线
        guard_result = input_guard.process(parsed_text, user_context)
        if guard_result.risk_level >= RiskLevel.RED:
            return reject("MALICIOUS_CONTENT_DETECTED")

    # ── 5. 文档级权限标记 ──
    document.add_acl(user_context.acl_tags)

    # ── 6. 内容审核 ──
    content_check = content_moderator.check(parsed_text)
    if content_check.is_unsafe:
        return reject("UNSAFE_CONTENT:" + content_check.category)

    # ── 7. 分块 + 向量化 + 入库 ──
    chunks = text_splitter.split(guard_result.sanitized_text)
    embeddings = embedder.embed(chunks)
    vector_store.insert(chunks, embeddings, metadata={
        "source": document.source,
        "acl": document.acl,
        "content_hash": sha256(parsed_text)
    })

    return success("DOCUMENT_INGESTED")

10.3 检索结果安全过滤

def rag_retrieval_filter(raw_results, user_context):
    filtered = []
    for doc in raw_results:
        # 1. ACL权限检查
        if not acl_check(doc.metadata["acl"], user_context):
            continue

        # 2. 内容安全检查
        if contains_malicious_content(doc.content):
            log_warning("MALICIOUS_DOC_DETECTED", doc.id)
            continue

        # 3. 来源标记
        doc.metadata["source_label"] = f"[来源: {doc.title}]"

        # 4. 截断（单文档最多500字符注入Prompt）
        doc.content = doc.content[:500]

        filtered.append(doc)

        # 5. 限制检索结果数量（最多5条）
        if len(filtered) >= 5:
            break

    return filtered

11.3 SafeToolWrapper执行流程

class AgentSafeWrapper:
    """Agent工具调用安全包装器"""

    async def execute(self, tool_name: str, params: dict, ctx: SecurityContext):
        # 1. 工具存在性与可用性
        tool = self.registry.get(tool_name)
        if not tool:
            return ToolResult.error("TOOL_NOT_FOUND")

        # 2. 参数Schema校验
        errors = validate_params(params, tool.params_schema)
        if errors:
            return ToolResult.error("PARAM_VALIDATION_FAILED", errors)

        # 3. 参数内容安全检查（防注入）
        for key, val in params.items():
            if isinstance(val, str):
                for forbidden in FORBIDDEN_PATTERNS:
                    if forbidden in val:
                        return ToolResult.error("FORBIDDEN_PATTERN_IN_PARAMS")

        # 4. 权限检查
        if not self.authz.check(ctx.user_id, tool.auth_level):
            audit_log("PERMISSION_DENIED", tool_name, ctx)
            return ToolResult.error("PERMISSION_DENIED")

        # 5. 调用次数限制
        if not self.rate_limiter.allow(ctx.user_id, tool_name):
            return ToolResult.error("RATE_LIMIT_EXCEEDED")

        # 6. 人工审批（如需要）
        if tool.requires_approval:
            approved = await self.request_approval(tool_name, params, ctx)
            if not approved:
                audit_log("APPROVAL_DENIED", tool_name, ctx)
                return ToolResult.error("APPROVAL_REQUIRED")

        # 7. 沙盒执行
        try:
            async with sandbox(timeout=30):
                result = await tool.execute(params)
        except TimeoutError:
            audit_log("TOOL_EXECUTION_TIMEOUT", tool_name, ctx)
            return ToolResult.error("EXECUTION_TIMEOUT")

        # 8. 结果安全过滤
        filtered = self.output_filter.filter(result)

        # 9. 审计日志
        audit_log("TOOL_EXECUTED", tool_name, ctx, result_summary(filtered))

        return ToolResult.success(filtered)

def process_multimodal_input(input_data, user_context):
    result = {"text_segments": [], "media_segments": []}

    # ── 文本部分：走标准L2管线 ──
    if input_data.text:
        text_result = input_guard.process(input_data.text, user_context)
        result["text_segments"].append(text_result)

    # ── 媒体部分：逐文件处理 ──
    for media_item in input_data.media:
        # 1. 文件类型白名单
        if media_item.type not in ALLOWED_MEDIA_TYPES:
            raise SecurityError(f"UNSUPPORTED: {media_item.type}")
        # 2. 文件大小校验
        if media_item.size > MAX_FILE_SIZE[media_item.type]:
            raise SecurityError("FILE_TOO_LARGE")

        # 3. 沙盒内处理
        with sandbox(timeout=30):
            handler = MEDIA_HANDLERS[media_item.type]
            extracted = handler.extract(media_item)
            # ImageHandler: OCR提取文字
            # AudioHandler: ASR语音转文字
            # VideoHandler: 抽帧 → 逐帧OCR
            # FileHandler: 文档解析

        # 4. 提取的文字走输入防护管线
        text_check = input_guard.process(extracted.text, user_context)
        result["media_segments"].append({
            "type": media_item.type,
            "extracted_text": text_check.sanitized_text,
            "risk_level": text_check.risk_level
        })

        # 5. 图片/视频额外：内容审核（色情/暴力检测）
        if media_item.type in ["image", "video_frame"]:
            vision_check = vision_moderator.check(media_item)
            if vision_check.is_unsafe:
                raise SecurityError(f"UNSAFE_VISUAL: {vision_check.category}")

    return result
